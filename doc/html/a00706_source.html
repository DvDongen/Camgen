749  bool evaluate_weight()
750  {
751  if(!(evaluate_phase_space_weight() and evaluate_helicity_weight() and evaluate_colour_weight()))
752  {
753  this->weight()=(value_type)0;
754  set_integrands(0);
755  return false;
756  }
757  this->weight()=ps_weight*hel_weight*col_weight;
758  value_type f=pb_conversion*symmetry_factor*ps_factor*hel_factor*col_factor;
