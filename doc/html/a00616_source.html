477  bool generate()
478  {
479  if(procs.size()==0)
480  {
481  this->weight()=(value_type)0;
482  return false;
483  }
484  value_type rho=rn_stream::throw_number();
485  sub_proc=procs.begin();
486  value_type r(0);
487  while(r<rho and sub_proc!=procs.end())
488  {
489  r+=(*sub_proc)->alpha;
490  ++sub_proc;
491  }
492  --sub_proc;
493  if(sub_proc==procs.end())
494  {
495  log(log_level::warning)<<CAMGEN_STREAMLOC<<"subprocess iterator overflow detected--no generation performed"<<endlog;
496  return false;
497  }
498  bool q=(*sub_proc)->generate();
499  this->integrand()=(*sub_proc)->integrand();
500  this->weight()=(*sub_proc)->weight()/(*sub_proc)->alpha;
501  if(update_counter!=0 and auto_proc_adapt!=0 and update_counter%auto_proc_adapt==0)
502  {
503  adapt_processes();
504  }
505  this->up_to_date=false;
506  return q;
507  }

670  void adapt_processes()
671  {
672  value_type p=process_adaptivity();
673  value_type norm(0);
674  MC_integral<value_type>proc_xsec;
675  value_type sigma=cross_section().value;
676  value_type q=process_threshold()*sigma/procs.size();
677  for(process_iterator it=procs.begin();it!=procs.end();++it)
678  {
679  proc_xsec=(*it)->cross_section();
680  if((proc_xsec.value+proc_xsec.error)>q)
681  {
682  (*it)->alpha=std::pow((*it)->cross_section().error,p);
683  norm+=(*it)->alpha;
684  }
685  else
686  {
687  std::cout<<"deleting process with xsec "<<proc_xsec<<" smaller than "<<q<<std::endl;
688  (*it)->alpha=(value_type)0;
689  }
690  }
691  for(process_iterator it=procs.begin();it!=procs.end();++it)
692  {
693  (*it)->alpha/=norm;
694  }
695  std::sort(procs.begin(),procs.end(),alpha_more);
696  process_iterator it=procs.begin();
697  while(it!=procs.end() and (*it)->alpha!=(value_type)0)
698  {
699  ++it;
700  }
701  if(it==procs.end() or it==procs.begin())
702  {
703  return;
704  }
705  std::size_t newsize=it-procs.begin();
706  process_iterator it2;
707  while(it!=procs.end())
708  {
709  algorithm.set_process((*it)->amplitude);
710  algorithm.remove_process();
711  delete (*it);
712  ++it;
713  }
714  procs.resize(newsize);
715  sub_proc=procs.begin();
716  }
