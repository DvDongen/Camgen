50  static generator_type* create_instance(typename CM_algorithm<model_t,N_in,N_out>::tree_iterator it,is_generator_tag is_tag,fs_generator_tag fs_tag)
51  {
52  bool backward_shat=(fs_tag==phase_space_generators::recursive_backward_shat);
53  typename initial_state_factory<model_t,N_in,rng_t>::generator_type* isgen=initial_state_factory<model_t,N_in,rng_t>::create_instance(is_tag,backward_shat);
54  if(isgen==NULL)
55  {
56  return NULL;
57  }
58  fs_generator_tag fst=fs_tag;
59  if(backward_shat and isgen->s_hat_sampling)
60  {
61  log(log_level::warning)<<CAMGEN_STREAMLOC<<"initial state does not allow s-hat sampling by final state generator--continuing with IS s-hat sampling"<<endlog;
62  fst=phase_space_generators::recursive_backward_s;
63  }
64  generator_type* result=NULL;
65  switch(fst)
66  {
67  case phase_space_generators::uniform:
68  result=rambo<model_t,N_in,N_out,rng_t>::create_instance(it,isgen);
69  break;
70  case phase_space_generators::recursive:
71  result=ps_tree<model_t,N_in,N_out,rng_t>::create_instance(it,isgen);
72  break;
73  case phase_space_generators::recursive_backward_s:
74  result=ps_tree<model_t,N_in,N_out,rng_t>::create_instance(it,isgen);
75  break;
76  case phase_space_generators::recursive_backward_shat:
77  result=ps_tree<model_t,N_in,N_out,rng_t>::create_instance(it,isgen);
78  break;
79  default:
80  break;
81  }
82  if(result!=NULL)
83  {
84  for(int i=1;i<=(int)N_in;++i)
85  {
86  result->set_beam_energy(-i,beam_energy(i));
87  }
88  result->refresh_Ecm();
89  for(int i=1;i<=(int)N_out;++i)
90  {
91  for(int j=i+1;j<=(int)N_out;++j)
92  {
93  result->set_m_min(i,j,basic_cuts::m_min(i,j));
94  }
95  }
96  result->refresh_m_min();
97  }
98  return result;
99  }

103  static generator_type* create_instance(typename CM_algorithm<model_t,N_in,N_out>::tree_iterator it,std::istream& is)
104  {
105  std::string initline,endline;
106  do
107  {
108  std::getline(is,initline);
109  }
110  while(initline!="<psgen>" and !is.eof());
111  if(is.eof())
112  {
113  log(log_level::warning)<<CAMGEN_STREAMLOC<<"end of file reached before initial data are read--returning NULL"<<endlog;
114  return NULL;
115  }
116  typename initial_state_factory<model_t,N_in,rng_t>::generator_type* isgen=initial_state_factory<model_t,N_in,rng_t>::create_instance(is);
117  if(isgen==NULL)
118  {
119  do
120  {
121  std::getline(is,endline);
122  }
123  while(endline!="</psgen>" and !is.eof());
124  return NULL;
125  }
126  do
127  {
128  std::getline(is,initline);
129  }
130  while(initline!="<fsgen>" and !is.eof());
131  if(is.eof())
132  {
133  log(log_level::warning)<<CAMGEN_STREAMLOC<<"end of file reached before initial data are read--returning NULL"<<endlog;
134  return NULL;
135  }
136  generator_type* result;
137  std::string fstype;
138  is>>fstype;
139  if(fstype=="rambo")
140  {
141  result=new rambo<model_t,N_in,N_out,rng_t>(isgen);
142  }
143  else if(fstype=="pstree")
144  {
145  bool q1=false,q2=false,q3=false;
146  typename generator_type::size_type b;
147  is>>q1>>q2>>q3>>b;
148  result=new ps_tree<model_t,N_in,N_out,rng_t>(isgen,false,q1,q2,q3,b);
149  }
150  else if(fstype=="pstree~")
151  {
152  bool q1=false,q2=false,q3=false;
153  typename generator_type::size_type b;
154  is>>q1>>q2>>q3>>b;
155  result=new ps_tree<model_t,N_in,N_out,rng_t>(isgen,true,q1,q2,q3,b);
156  }
157  else
158  {
159  log(log_level::warning)<<CAMGEN_STREAMLOC<<"momentum generator type "<<fstype<<" not recognised for "<<N_in<<" -> "<<N_out<<" process"<<endlog;
160  result=NULL;
161  }
162  if(result!=NULL)
163  {
164  if(!result->set_tree(it))
165  {
166  return NULL;
167  }
168  result->load(is);
169  result->refresh_Ecm();
170  result->refresh_m_min();
171  }
172  do
173  {
174  std::getline(is,endline);
175  }
176  while(endline!="</psgen>" and !is.eof());
177  return result;
178  }
179  };
