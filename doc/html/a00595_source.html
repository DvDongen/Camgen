85  static generator_type* create_instance(typename CM_algorithm<model_t,N_in,N_out>::tree_iterator it,generator_tag tag)
86  {
87  switch(tag)
88  {
89  case colour_generators::uniform:
90  return uniform_colours<value_type,N_in,N_out,rng_t,cc>::template create_instance<model_t>(it);
91  case colour_generators::summation:
92  return colour_summer<value_type,N_in,N_out,cc>::template create_instance<model_t>(it);
93  case colour_generators::flow_sampling:
94  if(static_eq<typename model_t::colour_treatment,adjoint>::value)
95  {
96  return adjoint_QCD<value_type,N_in,N_out,model_t::N_c,rng_t,cc>::template create_instance<model_t>(it);
97  }
98  else if(static_eq<typename model_t::colour_treatment,colour_flow>::value)
99  {
100  return colour_flow_QCD<value_type,N_in,N_out,model_t::N_c,rng_t,cc>::template create_instance<model_t>(it);
101  }
102  else
103  {
104  return NULL;
105  }
106  default:
107  return NULL;
108  }
109  }

114  static generator_type* create_instance(typename CM_algorithm<model_t,N_in,N_out>::tree_iterator it,std::istream& is)
115  {
116  std::string initline,endline;
117  while(initline!="<colgen>" and !is.eof())
118  {
119  std::getline(is,initline);
120  }
121  if(is.eof())
122  {
123  log(log_level::warning)<<CAMGEN_STREAMLOC<<"end of file reached before initial data are read"<<endlog;
124  return NULL;
125  }
126  generator_type* result;
127  std::string type;
128  is>>type;
129  if(type=="sum")
130  {
131  result=colour_summer<value_type,N_in,N_out,cc>::template create_instance<model_t>(it);
132  }
133  else if(type=="uniform")
134  {
135  result=uniform_colours<value_type,N_in,N_out,rng_t,cc>::template create_instance<model_t>(it);
136  }
137  else if(type=="qcd")
138  {
139  if(static_eq<typename model_t::colour_treatment,adjoint>::value)
140  {
141  result=adjoint_QCD<value_type,N_in,N_out,model_t::N_c,rng_t,cc>::template create_instance<model_t>(it);
142  }
143  else if(static_eq<typename model_t::colour_treatment,colour_flow>::value)
144  {
145  result=colour_flow_QCD<value_type,N_in,N_out,model_t::N_c,rng_t,cc>::template create_instance<model_t>(it);
146  }
147  else
148  {
149  result=NULL;
150  }
151  }
152  else
153  {
154  log(log_level::warning)<<CAMGEN_STREAMLOC<<"colour generator type "<<type<<" not recognised"<<endlog;
155  result=NULL;
156  }
157  if(result!=NULL)
158  {
159  result->load(is);
160  }
161  do
162  {
163  std::getline(is,endline);
164  }
165  while(endline!="</colgen>" and !is.eof());
166  return result;
167  }
168  };
