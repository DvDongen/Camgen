62  static s_generator<value_t,rng_t>* create_instance(std::istream& is,const value_t* mass,const value_t* width,const value_t* nu)
63  {
64  std::string initflag;
65  do
66  {
67  std::getline(is,initflag);
68  }
69  while(initflag!="<s_gen>" and !is.eof());
70  if(is.eof())
71  {
72  log(log_level::warning)<<CAMGEN_STREAMLOC<<"end of file reached before initial data are read--returning NULL"<<endlog;
73  return NULL;
74  }
75  std::string id;
76  is>>id;
77  bool adaptive=false;
78  if(id[id.size()-1]=='*')
79  {
80  adaptive=true;
81  id.resize(id.size()-1);
82  }
83  s_generator<value_t,rng_t>* sgen;
84  if(id=="Dd")
85  {
86  sgen=new Dd_s_generator<value_t,rng_t>(mass);
87  }
88  else if(id=="uni")
89  {
90  sgen=new uni_s_generator<value_t,rng_t>();
91  }
92  else if(id=="pl")
93  {
94  sgen=new pl_s_generator<value_t,rng_t>(mass,nu);
95  }
96  else if(id=="BW")
97  {
98  sgen=new BW_s_generator<value_t,rng_t>(mass,width);
99  }
100  else if(id=="invcosh")
101  {
102  sgen=new inv_cosh_y_generator<value_t,rng_t>(mass);
103  }
104  else
105  {
106  log(log_level::warning)<<CAMGEN_STREAMLOC<<"s-generator id "<<id<<" not recognised--returning NULL."<<endlog;
107  std::string endflag;
108  do
109  {
110  std::getline(is,endflag);
111  }
112  while(endflag!="</s_gen>" and !is.eof());
113  return NULL;
114  }
115  s_generator<value_t,rng_t>* result;
116  if(adaptive)
117  {
118  result=new adaptive_s_generator<value_t,rng_t>(sgen,500);
119  }
120  else
121  {
122  result=sgen;
123  }
124  result->load(is);
125  std::string endflag;
126  do
127  {
128  std::getline(is,endflag);
129  }
130  while(endflag!="</s_gen>" and !is.eof());
131  return result;
